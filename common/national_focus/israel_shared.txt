	cost = 10
	search_filters = {  FOCUS_FILTER_INDUSTRY }
	
	available_if_capitulated = yes
	available = {
		any_state = {
			is_coastal = yes
			is_controlled_by = ROOT
		}
	}
	
	completion_reward = {
		navy_experience = 25
		if = {
			limit = {
				
				any_owned_state = {
					is_coastal = yes
				}

				limit = {
					is_coastal = yes
					free_building_slots = {
						building = dockyard
						size > 2

					}
				}
				add_extra_state_shared_building_slots = 3
				add_building_construction = {
					type = dockyard
					level = 3

				}
			}
		}
	}			
}
